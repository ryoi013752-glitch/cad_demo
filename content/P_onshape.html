import cmath
import math

# Solution 3:
def t1_sol3(x, y):
    rad = 2.0 * cmath.atan(
        (1280.0 * y + cmath.sqrt(
            -4000000.0 * x**4
            - 3200000.0 * x**3
            - 8000000.0 * x**2 * y**2
            - 4000000.0 * x**2 * y
            + 178400.0 * x**2
            - 3200000.0 * x * y**2
            - 1600000.0 * x * y
            + 327360.0 * x
            - 4000000.0 * y**4
            - 4000000.0 * y**3
            - 181600.0 * y**2
            + 409200.0 * y
            + 125911.0
        ) + 320.0) / (
            2000.0 * x**2 + 2080.0 * x + 2000.0 * y**2 + 1000.0 * y + 461.0
        )
    )
    return math.degrees(rad.real)

def t2_sol3(x, y):
    rad = 2.0 * cmath.atan(
        (1280.0 * y - cmath.sqrt(
            -4000000.0 * x**4
            + 3200000.0 * x**3
            - 8000000.0 * x**2 * y**2
            - 4000000.0 * x**2 * y
            + 178400.0 * x**2
            + 3200000.0 * x * y**2
            + 1600000.0 * x * y
            - 327360.0 * x
            - 4000000.0 * y**4
            - 4000000.0 * y**3
            - 181600.0 * y**2
            + 409200.0 * y
            + 125911.0
        ) + 320.0) / (
            2000.0 * x**2 + 480.0 * x + 2000.0 * y**2 + 1000.0 * y - 51.0
        )
    )
    return math.degrees(rad.real)

# 測試
print(t1_sol3(0.2, 0.185), t2_sol3(0.2, 0.185))
